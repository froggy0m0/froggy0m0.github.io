```python
def solution(board,moves):
    result = list() # 바구니 ?
    cnt = 0
    for j in moves:
        for i in range(len(board)):
            if board[i][j-1] != 0 :
                result.append(board[i][j-1])
                board[i][j-1] = 0
                break
        if len(result) >= 2 and result[-1] == result[-2]:
            result = result[:-2]
            cnt +=2

    return cnt
```
