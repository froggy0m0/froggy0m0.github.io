```python
class Person:
    def __init__(self):
        self.ht = False
        self.pump = False
        self.pen = False
        self.led = False

    def ht_switch(self):
        if self.ht :
            self.ht = False
            print("ht를 off")
        else:
            self.ht = True
            print("ht를 on")

    def pump_switch(self):
        if self.pump :
            self.pump = False
            print("pump를 off")
        else:
            self.pump = True
            print("pump를 on")

    def pen_switch(self):
        if self.pen :
            self.pen = False
            print("pen를 off")
        else:
            self.pen = True
            print("pen를 on")

    def led_switch(self):
        if self.led :
            self.led = False
            print("led를 off")
        else:
            self.led = True
            print("led를 on")


    def ht_status(self):
        print("ht는 on상태입니다" if self.ht else "ht는 off상태입니다")
    def pump_status(self):
        print("pump는 on상태입니다" if self.pump else "pump는 off상태입니다")
    def pen_status(self):
        print("pen는 on상태입니다" if self.pen else "pump는 off상태입니다")
    def led_status(self):
        print("led는 on상태입니다" if self.led else "led는 off상태입니다")

```

import person

pp = person.Person()
