### table

```mysql
create table info(
pp VARCHAR(20) NOT NULL,
id VARCHAR(20) NOT NULL,
dd datetime,
PRIMARY KEY(pp)
);
```



### insert

```python
import pymysql
import datetime
import time

d = str(datetime.datetime.today())


db =pymysql.connect(
    host='127.0.0.1',
    port=3306,
    user='root',
    passwd='12345',
    db='plant',
    charset='utf8'
)

cursor = db.cursor()

sql = "select pp from info order by dd DESC limit 1"

cursor.execute(sql)

res = cursor.fetchone()
print(res)
if res == None :
    pp = '0x00'
else :
    res = res[0]
    pp = int(res,16)+1
    pp = hex(pp)

print(pp)
sql = 'insert into info values(%s,%s,%s);'

cursor.execute(sql,(pp,'jj001',d))

res = cursor.fetchone()

db.commit()

time.sleep(2)

db.close()
```

### select

```python
import pymysql
import time


db =pymysql.connect(
    host='127.0.0.1',
    port=3306,
    user='root',
    passwd='12345',
    db='plant',
    charset='utf8'
)

cursor = db.cursor()

sql = "select * from info where id = 'jj001' order by dd  Desc limit 5;"

cursor.execute(sql)

res = cursor.fetchall()
for i in res:
	print(i[0],i[1],i[2])


db.commit()

time.sleep(2)

db.close()
```
